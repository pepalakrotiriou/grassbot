Βεβαιώσου ότι η IP είναι η σωστή! const PI_IP = "http://192.168.1.15:5000"; const MODEL_URL = "https://teachablemachine.withgoogle.com/models/c6m5BlOJg/"; let model; async function init() { try { // Φόρτωση μοντέλου model = await tmImage.load( MODEL_URL + "model.json", MODEL_URL + "metadata.json" ); document.getElementById("result-box").innerHTML = "✅ Το σύστημα είναι έτοιμο!"; } catch(e) { console.error(e); document.getElementById("result-box").innerHTML = "❌ Σφάλμα φόρτωσης μοντέλου."; } } async function predict() { if (!model) return; const img = document.getElementById("image-display"); // Προσθήκη ελέγχου αν η εικόνα έχει όντως δεδομένα if (img.src.includes("data:image/gif;base64")) return; try { const prediction = await model.predict(img); prediction.sort((a, b) => b.probability - a.probability); const res = prediction[0]; const color = res.className.toLowerCase().includes("grass") ? "#4ade80" : "#ff4d4d"; document.getElementById("result-box").style.borderLeftColor = color; document.getElementById("result-box").innerHTML = `Διάγνωση: ${res.className} (${(res.probability * 100).toFixed(0)}%)`; } catch (err) { console.error("Prediction error:", err); } } // Διόρθωση στην captureFromPi για να αποφύγουμε το caching του browser async function captureFromPi() { const img = document.getElementById("image-display"); document.getElementById("result-box").innerHTML = "⏳ Λήψη από κάμερα..."; // Το ?t= διασφαλίζει ότι ο browser θα ζητήσει νέα φωτό και όχι την παλιά από τη μνήμη img.src = `${PI_IP}/capture?t=${new Date().getTime()}`; }
